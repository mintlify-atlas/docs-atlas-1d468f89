## Overview

Message component displays inline messages with different severity levels.

## Import

```typescript
import { MagaryMessage } from 'ng-magary';
```

## Basic Usage

```html
<magary-message
  [severity]="'success'"
  [text]="'Operation completed successfully'">
</magary-message>
```

## Conditional Display

<CodeGroup>
```html Component
<magary-message
  *ngIf="showMessage"
  [severity]="'success'"
  [text]="'Form submitted successfully'">
</magary-message>

<form (ngSubmit)="onSubmit()">
  <!-- form fields -->
  <button type="submit">Submit</button>
</form>
```

```typescript TypeScript
import { Component } from '@angular/core';
import { MagaryMessage } from 'ng-magary';
import { CommonModule } from '@angular/common';

@Component({
  selector: 'app-example',
  standalone: true,
  imports: [MagaryMessage, CommonModule],
  templateUrl: './example.component.html'
})
export class ExampleComponent {
  showMessage = false;
  
  onSubmit() {
    // Submit logic
    this.showMessage = true;
    
    // Hide after 3 seconds
    setTimeout(() => {
      this.showMessage = false;
    }, 3000);
  }
}
```
</CodeGroup>

## Input Properties

<ParamField path="severity" type="'success' | 'info' | 'warn' | 'error' | 'secondary' | 'contrast'" default="'info'">

<ParamField path="text" type="string">

<ParamField path="detail" type="string">

<ParamField path="onClose" type="EventEmitter<Event>">

## Complete Example

<CodeGroup>
```html Component
<div class="messages-container">
  <magary-message
    *ngFor="let msg of messages"
    [severity]="msg.severity"
    [text]="msg.text"
    [detail]="msg.detail"
    [closable]="true"
    (onClose)="removeMessage(msg)">
  </magary-message>
</div>

<button (click)="addSuccess()">Add Success</button>
<button (click)="addError()">Add Error</button>
```

```typescript TypeScript
import { Component } from '@angular/core';
import { MagaryMessage } from 'ng-magary';
import { CommonModule } from '@angular/common';

interface Message {
  severity: 'success' | 'error' | 'warn' | 'info';
  text: string;
  detail?: string;
}

@Component({
  selector: 'app-example',
  standalone: true,
  imports: [MagaryMessage, CommonModule],
  templateUrl: './example.component.html'
})
export class ExampleComponent {
  messages: Message[] = [];
  
  addSuccess() {
    this.messages.push({
      severity: 'success',
      text: 'Success',
      detail: 'Operation completed successfully'
    });
  }
  
  addError() {
    this.messages.push({
      severity: 'error',
      text: 'Error',
      detail: 'Something went wrong'
    });
  }
  
  removeMessage(msg: Message) {
    this.messages = this.messages.filter(m => m !== msg);
  }
}
```
</CodeGroup>

## Features

- **Multiple severities**: 6 severity levels
- **Closable**: Optional close button
- **Auto-dismiss**: Configurable timeout
- **Custom icons**: Override default icons
- **Animations**: Smooth enter/exit
- **Detailed text**: Summary and detail support
- **Inline display**: Shows within page flow

## Accessibility

- ARIA role="alert" for important messages
- Semantic severity colors
- Keyboard accessible close button
- Screen reader support
