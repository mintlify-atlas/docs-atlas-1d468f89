## Overview

Avatar component displays user profile images, initials, or icons with support for badges and various shapes.

## Import

```typescript
import { MagaryAvatar } from 'ng-magary';
```

## Basic Usage

### Image Avatar

```html
<magary-avatar
  [image]="'assets/profile.jpg'"
  [alt]="'User profile'">
</magary-avatar>
```

### Label Avatar

```html
<magary-avatar
  [label]="'John Doe'">
</magary-avatar>
```

### Icon Avatar

```html
<magary-avatar
  [icon]="'user'">
</magary-avatar>
```

## Clickable Avatar

<CodeGroup>
```html Component
<magary-avatar
  [image]="'assets/profile.jpg'"
  [clickable]="true"
  (avatarClick)="handleAvatarClick($event)">
</magary-avatar>
```

```typescript TypeScript
import { Component } from '@angular/core';
import { MagaryAvatar, AvatarClickEvent } from 'ng-magary';

@Component({
  selector: 'app-example',
  standalone: true,
  imports: [MagaryAvatar],
  templateUrl: './example.component.html'
})
export class ExampleComponent {
  handleAvatarClick(event: AvatarClickEvent) {
    if (event.type === 'avatar') {
      console.log('Avatar clicked');
      // Navigate to profile
    } else if (event.type === 'badge') {
      console.log('Badge clicked');
      // Show notifications
    }
  }
}
```
</CodeGroup>

## Input Properties

<ParamField path="label" type="string">
  Text label (displays initials automatically)
</ParamField>

<ParamField path="image" type="string">
  Image URL
</ParamField>

<ParamField path="icon" type="string">
  Lucide icon name
</ParamField>

<ParamField path="shape" type="'circle' | 'square'" default="'square'">
  Shape of the avatar
</ParamField>

<ParamField path="size" type="'xsmall' | 'small' | 'normal' | 'large' | 'xlarge'" default="'normal'">
  Size of the avatar
</ParamField>

<ParamField path="badgeValue" type="string | number">
  Value to display in badge
</ParamField>

<ParamField path="badgeSeverity" type="'info' | 'success' | 'warning' | 'danger'" default="'danger'">
  Severity level of badge
</ParamField>

<ParamField path="alt" type="string" default="'Avatar'">
  Alt text for image
</ParamField>

<ParamField path="clickable" type="boolean" default="false">
  Make avatar clickable
</ParamField>

<ParamField path="loading" type="boolean" default="false">
  Show loading state
</ParamField>

<ParamField path="disabled" type="boolean" default="false">
  Disable avatar interaction
</ParamField>

<ParamField path="customStyle" type="Record<string, unknown>">
  Custom inline styles
</ParamField>

## Output Events

<ParamField path="avatarClick" type="EventEmitter<AvatarClickEvent>">
  Emitted when avatar or badge is clicked
</ParamField>

## Interfaces

```typescript
interface AvatarClickEvent {
  type: 'avatar' | 'badge';
  data?: {
    label?: string;
    image?: string;
    icon?: string;
    value?: string | number;
    severity?: BadgeSeverity;
  };
}

type AvatarSize = 'xsmall' | 'small' | 'normal' | 'large' | 'xlarge';
type AvatarShape = 'circle' | 'square';
type BadgeSeverity = 'info' | 'success' | 'warning' | 'danger';
```

## Features

- **Multiple types**: Image, label (initials), or icon
- **Automatic initials**: Extracts initials from label
- **Badge support**: Notification badges
- **5 sizes**: From xsmall to xlarge
- **2 shapes**: Circle and square
- **Clickable**: Optional click handlers
- **Loading state**: Visual loading indicator
- **Image fallback**: Falls back to label/icon on error
